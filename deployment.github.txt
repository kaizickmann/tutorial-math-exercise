
# Create Github repository

https://github.com/kaizickmann/tutorial-math-exercise.git


# Install Angular CLI gh-pages

npm i angular-cli-ghpages --save-dev


# Run build

npm run build:github

// ng build --configuration production --base-href "https://kaizickmann.github.io/tutorial-math-exercise/"
// ng build --prod                     --base-href "https://GithubUserName.github.io/GithubRepoName/"


# Deploy to gh-pages

npx angular-cli-ghpages --dir=dist/math-exercise

https://kaizickmann.github.io/tutorial-math-exercise/
